Comandos Rails

#cria um novo projeto
rails new loja

#inicia server
rails server ou s

#gera um controller
rails g controller Produtos

#cria banco de dados
rails db:create

#cria um model
rails g model Produto nome:string descricao:text preco:decimal quantidade:integer

#faz migração
rails db:migrate

#acessa o console db
rails dbconsole

#instala dependencias
bundle install

#instala bootstrap
rails generate bootstrap:install static

#cria crud completo
rails g scaffold Departamento nome
